#include <iostream>
using namespace std;

int main(){
	int nilai[] = {3,5,6,1,4,9,8,10,7,2};
	int length = sizeof(nilai)/sizeof(*nilai);
	int temp, cariData, hasilCari;
	
	for(int a = 1; a < length; a++){
		for(int b = 0; b < length-a; b++){
			if(nilai[b] > nilai[b+1]){
				temp = nilai[b];
				nilai[b] = nilai[b+1];
				nilai[b+1] = temp;
			}
		}
	}
	
	cout << "HASIL SORTING : " << endl;
	for(int c = 0; c < length; c++){
		cout << "INDEX KE-" << c << ": " << nilai[c] << endl;
	}
	
	cout << endl;
	cout << "CARI DATA : ";
	cin >> cariData;
	for(int d = 0; d < length; d++){
		if(cariData == nilai[d]){
			hasilCari++;
		}
	}
	if(hasilCari == 0){
		cout << "DATA TIDAK DITEMUKAN!" << endl;
	} else {
		for(int e = 0; e < length; e++){
			if(cariData == nilai[e]){
				cout << "DATA " << cariData << " ditemukan di Index ke-" << e << endl;
			}
		}
	}
}
